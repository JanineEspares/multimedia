Project Overview

A 3D solar system environment built using Blender for asset layout, texture mapping, and rendering. This project features planetary models mapped with high-definition textures and animated using custom cyclic paths. The scene is enhanced with Python scripting to procedurally distribute a 300-piece asteroid belt and deploy frame-dependent math expressions for object rotations, demonstrating practical skills in both 3D environment design and programmatic automation.

def get_cinematic_vectors(p_name, frame, settings):
    p_obj = bpy.data.objects.get(p_name)
    if not p_obj: return Vector((0,0,0)), Vector((0,0,0))
    p_scale = max(p_obj.scale)
    
    bpy.context.scene.frame_set(int(frame))
    pos = p_obj.matrix_world.to_translation()
    
    bpy.context.scene.frame_set(int(frame) - 2)
    pos_prev = p_obj.matrix_world.to_translation()
    
    vel = (pos - pos_prev)
    if vel.length < 0.0001: vel = Vector((-pos.y, pos.x, 0)) 
    vel = vel.normalized()
    
    to_sun = (sun_center - pos).normalized()
    if to_sun.length < 0.0001: to_sun = Vector((0, -1, 0))
    
    if p_name == "Venus":
        cam_dir = (-to_sun * 1.2 - vel * 1.5).normalized()
    else:
        cam_dir = (to_sun * 1.2 - vel * 1.0).normalized()
        
    cam_pos = pos + cam_dir * (settings["dist"] * p_scale) + Vector((0, 0, settings["z_offset"] * p_scale))
    return pos, cam_pos

        if "shuttle" in obj.name.lower():
            shuttle_obj = obj
            break

if shuttle_obj:
    transition_duration = int(fps * 2.5)   # 2.5-second smooth pan to shuttle
    shuttle_follow_dur = int(fps * 10.0)   # Clean, locked 10-second tracking window

    transition_start = current_frame
    transition_end = transition_start + transition_duration
    hero_end = transition_end + shuttle_follow_dur

    hero_offset = Vector((-1.5, -3.0, 4.0))

    # --- Phase 1: Pluto to Shuttle High-Angle Transition ---
    bpy.context.scene.frame_set(transition_start)
    shuttle_loc = shuttle_obj.matrix_world.to_translation()
    target_empty.location = shuttle_loc
    
    target_empty.keyframe_insert(data_path="location", frame=transition_start)
    tour_cam.keyframe_insert(data_path="location", frame=transition_start)
    cam_data.keyframe_insert(data_path="lens", frame=transition_start)

    bpy.context.scene.frame_set(transition_end)
    shuttle_loc = shuttle_obj.matrix_world.to_translation()
    target_empty.location = shuttle_loc
    tour_cam.location = shuttle_loc + hero_offset
    cam_data.lens = 45 # Tightened framing zoom

    target_empty.keyframe_insert(data_path="location", frame=transition_end)
    tour_cam.keyframe_insert(data_path="location", frame=transition_end)
    cam_data.keyframe_insert(data_path="lens", frame=transition_end)

    # --- Phase 2: Complete 10-Second Semi-Eagle Eye Tracking Engine ---
    for f in range(transition_end + 2, hero_end + 1, 2):
        bpy.context.scene.frame_set(f)
        s_pos_curr = shuttle_obj.matrix_world.to_translation()
        target_empty.location = s_pos_curr
        tour_cam.location = s_pos_curr + hero_offset
        
        target_empty.keyframe_insert(data_path="location", frame=f)
        tour_cam.keyframe_insert(data_path="location", frame=f)
        
    current_frame = hero_end

# PART 7: SINGLE GRAND SOLAR SYSTEM REVEAL (GLITCH-FREE FIXED LERP)

reveal_duration = int(fps * 8.0)   # 8-second smooth continuous pullback
reveal_start = current_frame
reveal_end = reveal_start + reveal_duration

bpy.context.scene.frame_set(reveal_start)
start_cam_loc = tour_cam.location.copy()
start_target_loc = target_empty.location.copy() 
end_cam_loc = sun_center + Vector((0, -45, 18))
start_lens = cam_data.lens
end_lens = 22 

for f in range(reveal_start, reveal_end + 1, 2):
    t = (f - reveal_start) / reveal_duration
    bpy.context.scene.frame_set(f)
    
    tour_cam.location = start_cam_loc.lerp(end_cam_loc, t)
    target_empty.location = start_target_loc.lerp(sun_center, t)  
    cam_data.lens = start_lens + (end_lens - start_lens) * t
    
    tour_cam.keyframe_insert(data_path="location", frame=f)
    target_empty.keyframe_insert(data_path="location", frame=f)
    cam_data.keyframe_insert(data_path="lens", frame=f)

current_frame = reveal_end 
bpy.context.scene.frame_set(1)
bpy.context.scene.frame_end = current_frame


# Global Bezier Spline Interp System
def apply_smooth_bezier(obj):
    if obj.animation_data and obj.animation_data.action:
        action = obj.animation_data.action
        fcurves = []
        if hasattr(action, "fcurves"):
            fcurves = action.fcurves
        elif hasattr(action, "layers"):
            for layer in action.layers:
                if hasattr(layer, "channels"):
                    for channel in layer.channels:
                        if hasattr(channel, "fcurves"):
                            fcurves.extend(channel.fcurves)
        for fcurve in fcurves:
            for kp in fcurve.keyframe_points:
                if kp.frame >= transition_end:
                    kp.interpolation = 'LINEAR'
                else:
                    kp.interpolation = 'BEZIER'
                    kp.handle_left_type = 'AUTO'
                    kp.handle_right_type = 'AUTO'

for item in [tour_cam, target_empty, cam_data]:
    apply_smooth_bezier(item)

m_links.new(m_coord.outputs['Generated'], m_separate.inputs['Vector'])

m_ramp.color_ramp.elements[0].position = 0.0

m_mid.color = (1.0, 0.25, 0.01, 1.0) 
m_ramp.color_ramp.elements[-1].position = 1.0
m_ramp.color_ramp.elements[-1].color = (1.0, 0.95, 0.88, 1.0) 

m_math_brightness.operation = 'MULTIPLY'

m_links.new(m_obj_info.outputs['Random'], m_math_brightness.inputs[0])

m_links.new(m_ramp.outputs['Color'], m_principled.inputs['Base Color'])
m_links.new(m_ramp.outputs['Color'], m_principled.inputs['Emission Color'])
m_links.new(m_math_brightness.outputs['Value'], m_principled.inputs['Emission Strength'])
m_links.new(m_principled.outputs['BSDF'], m_output.inputs['Surface'])

m_mesh = bpy.data.meshes.new("Meteor_Base_Mesh")
m_verts = [
    Vector((0, 0.1, 0)),         
    Vector((0.015, 0, 0.015)),   
    Vector((-0.015, 0, 0.015)),

m_faces = [(0,1,2), (0,2,3), (0,3,4), (0,4,1), (5,2,1), (5,3,2), (5,4,3), (5,1,4)]

# --- Region A: Atmospheric Entry ---
for planet_name in ["Earth", "Mars"]:

    if not planet_obj: continue

        spawn_angle = random.uniform(0, 2 * math.pi)
        local_radius = random.uniform(1.2, 1.6)
        
        start_x = math.cos(spawn_angle) * local_radius
        start_y = math.sin(spawn_angle) * local_radius
        start_z = random.uniform(-0.1, 0.1)
        
        entry_lifespan = random.randint(35, 60)
        vel_x = (-math.sin(spawn_angle) * 0.5 - start_x * 0.2) / entry_lifespan
        vel_y = (math.cos(spawn_angle) * 0.5 - start_y * 0.2) / entry_lifespan
        vel_z = random.uniform(-0.005, 0.005)

            drv.expression = f"{start_val:.4f} + ((frame + {timeline_offset}) % {entry_lifespan}) * {vel_val:.4f}"
            
        sf = random.uniform(0.12, 0.35)
        met_streak.scale = (sf, sf, sf * random.uniform(0.6, 1.8))

Project Specifications

Development Stack

Blender 5.1.2: 3D asset generation, environments, and physics keyframing.

Python API: Procedural scene automation and programmatic layout control.

Core Features

  • Animation Drivers: Mathematical math expressions automate object spin and orbital paths without keyframes.
  • Procedural Asteroid Belt: 300 unique randomized meshes orbiting via customized sine/cosine math.
  • Path-Constrained Orbiter: Multi-segmented space shuttle bound to a 200-frame cyclic Bezier curve path.
  • Cinematic Camera Rig: Dynamic tracking empty adjusting focal lengths based on target velocity vectors.

Animation Sequence

  • Frames 1–50: System-wide establishing shot tracking orbital initialization.
  • Frames 51–150: Mid-sequence focus shift tracking the path-constrained shuttle vector.
  • Frames 151–200: Loop closure capturing camera rig alignment adjustments.

Surface Assets: High-definition wraps via the Solar System Textures.
